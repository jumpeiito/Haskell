\pagestyle{empty}
\newcounter{width}
\newcounter{height}
\newcounter{tempw}
\newcounter{temph}
\newlength{\templength}
\setcounter{width}{0}
\setcounter{height}{0}
\def\resizeString#1#2{
  \settowidth{\templength}{#2}
  \ifdim\templength > #1
  \resizebox{#1}{10pt}{#2}
  \else
  #2
  \fi
}
\def\resetcounter{
  \ifnum\thewidth=140
  \setcounter{width}{0}
  \addtocounter{height}{42}
  \else
  \addtocounter{width}{70}
  \fi
}
\newcommand{\personallabel}[4]{%
  \setcounter{temph}{\value{height}}
  \setcounter{tempw}{\value{width}}
  \addtocounter{temph}{33}
  \addtocounter{tempw}{7}
  \put(\thetempw,\thetemph){〒#1} % 郵便番号
  \addtocounter{temph}{-9}
  \put(\thetempw,\thetemph){\resizeString{57mm}{#2}} % 住所1
  \addtocounter{tempw}{3}
  \addtocounter{temph}{-7}
  \put(\thetempw,\thetemph){\resizeString{52mm}{#3}} % 住所2
  \addtocounter{tempw}{-7}
  \addtocounter{temph}{-9}
  \put(\thetempw,\thetemph){\makebox[70mm][c]{\bf #4~様}} % 氏名
  \resetcounter
}
